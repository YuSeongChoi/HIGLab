✨ Visual Intelligence?

Visual Intelligence is a groundbreaking visual AI framework introduced in iOS 18. Long-press the Camera Control button to analyze your surroundings in real-time — object recognition, text extraction, place info, product search and more. Integrated with Apple Intelligence, it processes quickly and accurately on-device while fully protecting user privacy.

💡 Key Features: Real-time Object Recognition · Text Extraction (OCR) · Place/Landmark Recognition · Product Search · QR/Barcode Scanning · Translation · On-Device Processing · Camera Control Integration

📸 1. Basic Setup

Initialize the Visual Intelligence service and request permissions.

VisualIntelligenceManager.swift — Basic Setup
import SwiftUI
import AVFoundation

// Visual Intelligence 서비스 관리자
@Observable
class VisualIntelligenceManager {
    var isAuthorized = false
    var isAnalyzing = false
    var currentResult: AnalysisResult?

    // 카메라 권한 요청
    func requestAuthorization() async -> Bool {
        let status = await AVCaptureDevice.requestAccess(for: .video)
        await MainActor.run {
            isAuthorized = status
        }
        return status
    }

    // Visual Intelligence 지원 확인
    func isSupported() -> Bool {
        // iOS 18+ 및 Camera Control 지원 기기
        if #available(iOS 18.0, *) {
            return true
        }
        return false
    }
}

// 분석 결과 타입
struct AnalysisResult: Identifiable {
    let id = UUID()
    let type: AnalysisType
    let data: Any
    let timestamp = Date()
}

enum AnalysisType {
    case objectDetection
    case textExtraction
    case placeRecognition
    case productSearch
    case barcodeScanning
}

                } else {
                    ForEach(detectedObjects) { obj in
                        HStack {
                            Image(systemName: obj.icon)
                                .foregroundStyle(.blue)
                            VStack(alignment: .leading) {
                                Text(obj.label)
                                    .font(.subheadline)
                                Text("\(Int(obj.confidence * 100))% 신뢰도")
                                    .font(.caption)
                                    .foregroundStyle(.secondary)
                            }
                            Spacer()
                            Button("검색") {
                                searchObject(obj)
                            }
                            .buttonStyle(.bordered)
                        }
                        .padding()
                        .background(Color.gray.opacity(0.1))
                        .cornerRadius(8)
                    }
                }
            }
            .padding()
        }
        .task {
            await manager.requestAuthorization()
        }
    }

    func searchObject(_ object: DetectedObject) {
        print("🔍 검색: \(object.label)")
    }
}

// 감지된 객체 모델
struct DetectedObject: Identifiable {
    let id = UUID()
    let label: String
    let confidence: Double
    let bounds: CGRect

    var icon: String {
        // 객체 타입에 따라 아이콘 반환
        switch label.lowercased() {
        case let l where l.contains("person"): return "person.fill"
        case let l where l.contains("car"): return "car.fill"
        case let l where l.contains("dog"): return "pawprint.fill"
        case let l where l.contains("cat"): return "cat.fill"
        default: return "cube.fill"
        }
    }
}

📝 3. 텍스트 추출 (OCR)

Recognize and extract text in images — copy, translate, and search.

TextExtractionView.swift — Text Extraction
import SwiftUI
import Vision

@Observable
class TextExtractionManager {
    var extractedText: [RecognizedText] = []
    var isProcessing = false

    // 텍스트 추출 (Live Text)
    func extractText(from image: UIImage) async {
        await MainActor.run { isProcessing = true }

        guard let cgImage = image.cgImage else { return }

        let request = VNRecognizeTextRequest { request, error in
            guard error == nil,
                  let observations = request.results as? [VNRecognizedTextObservation]
            else { return }

            let recognizedTexts = observations.compactMap { observation in
                guard let topCandidate = observation.topCandidates(1).first else {
                    return nil
                }

                return RecognizedText(
                    text: topCandidate.string,
                    confidence: topCandidate.confidence,
                    bounds: observation.boundingBox
                )
            }

            Task { @MainActor in
                self.extractedText = recognizedTexts
                self.isProcessing = false
            }
        }

        // 정확도 설정
        request.recognitionLevel = .accurate
        request.usesLanguageCorrection = true

        let handler = VNImageRequestHandler(cgImage: cgImage, options: [:])
        try? handler.perform([request])
    }
}

struct RecognizedText: Identifiable {
    let id = UUID()
    let text: String
    let confidence: Float
    let bounds: CGRect
}

struct TextExtractionView: View {
    @State private var manager = TextExtractionManager()
    @State private var selectedImage: UIImage?
    @State private var showImagePicker = false

    var body: some View {
        ScrollView {
            VStack(spacing: 20) {
                // 이미지 선택
                if let selectedImage {
                    Image(uiImage: selectedImage)
                        .resizable()
                        .scaledToFit()
                        .cornerRadius(12)
                } else {
                    Button {
                        showImagePicker = true
                    } label: {
                        VStack {
                            Image(systemName: "photo.badge.plus")
                                .font(.system(size: 50))
                            Text("이미지 선택")
                        }
                        .frame(height: 200)
                        .frame(maxWidth: .infinity)
                        .background(Color.gray.opacity(0.1))
                        .cornerRadius(12)
                    }
                }

                // 추출된 텍스트
                if manager.isProcessing {
                    ProgressView("텍스트 추출 중...")
                        .padding()
                } else if !manager.extractedText.isEmpty {
                    VStack(alignment: .leading, spacing: 12) {
                        Text("추출된 텍스트")
                            .font(.headline)

                        ForEach(manager.extractedText) { item in
                            HStack {
                                Text(item.text)
                                    .textSelection(.enabled)

                                Spacer()

                                Menu {
                                    Button {
                                        copyText(item.text)
                                    } label: {
                                        Label("복사", systemImage: "doc.on.doc")
                                    }

                                    Button {
                                        translateText(item.text)
                                    } label: {
                                        Label("번역", systemImage: "translate")
                                    }

                                    Button {
                                        searchText(item.text)
                                    } label: {
                                        Label("검색", systemImage: "magnifyingglass")
                                    }
                                } label: {
                                    Image(systemName: "ellipsis.circle")
                                }
                            }
                            .padding()
                            .background(Color.gray.opacity(0.1))
                            .cornerRadius(8)
                        }
                    }
                }
            }
            .padding()
        }
        .onChange(of: selectedImage) { _, newImage in
            if let newImage {
                Task {
                    await manager.extractText(from: newImage)
                }
            }
        }
    }

    func copyText(_ text: String) {
        UIPasteboard.general.string = text
    }

    func translateText(_ text: String) {
        print("🌐 번역: \(text)")
    }

    func searchText(_ text: String) {
        print("🔍 검색: \(text)")
    }
}

📍 4. 장소 인식

Recognize landmarks, buildings, and places with relevant information.

PlaceRecognitionView.swift — Place Recognition
import SwiftUI
import CoreLocation
import MapKit

struct RecognizedPlace: Identifiable {
    let id = UUID()
    let name: String
    let category: String
    let address: String
    let coordinate: CLLocationCoordinate2D
    let rating: Double?
    let photoURL: URL?
}

@Observable
class PlaceRecognitionManager {
    var recognizedPlace: RecognizedPlace?
    var isRecognizing = false

    // 장소 인식
    func recognizePlace(from image: UIImage, location: CLLocation?) async {
        isRecognizing = true

        // 실제로는 Vision + Maps API 사용
        try? await Task.sleep(for: .seconds(1))

        // 예시 데이터
        recognizedPlace = RecognizedPlace(
            name: "애플 파크",
            category: "랜드마크",
            address: "One Apple Park Way, Cupertino, CA",
            coordinate: CLLocationCoordinate2D(latitude: 37.3349, longitude: -122.0090),
            rating: 4.8,
            photoURL: nil
        )

        isRecognizing = false
    }
}

struct PlaceRecognitionView: View {
    @State private var manager = PlaceRecognitionManager()
    @State private var cameraPosition: MapCameraPosition = .automatic

    var body: some View {
        VStack(spacing: 20) {
            // 카메라 뷰 (실제로는 AVCaptureSession)
            Rectangle()
                .fill(Color.black)
                .aspectRatio(3/4, contentMode: .fit)
                .cornerRadius(12)
                .overlay {
                    if manager.isRecognizing {
                        ProgressView("장소 인식 중...")
                            .tint(.white)
                    }
                }

            // 인식된 장소 정보
            if let place = manager.recognizedPlace {
                VStack(alignment: .leading, spacing: 16) {
                    // 장소 정보
                    VStack(alignment: .leading, spacing: 8) {
                        HStack {
                            Text(place.name)
                                .font(.title2)
                                .fontWeight(.bold)

                            Spacer()

                            if let rating = place.rating {
                                HStack(spacing: 4) {
                                    Image(systemName: "star.fill")
                                        .foregroundStyle(.yellow)
                                    Text("\(rating, specifier: "%.1f")")
                                        .fontWeight(.semibold)
                                }
                            }
                        }

                        Text(place.category)
                            .font(.subheadline)
                            .foregroundStyle(.secondary)

                        Label(place.address, systemImage: "mappin.circle")
                            .font(.footnote)
                    }
                    .padding()
                    .background(Color.gray.opacity(0.1))
                    .cornerRadius(12)

                    // 지도
                    Map(position: $cameraPosition) {
                        Marker(place.name, coordinate: place.coordinate)
                    }
                    .frame(height: 200)
                    .cornerRadius(12)

                    // 액션 버튼
                    HStack(spacing: 12) {
                        Button {
                            openInMaps(place)
                        } label: {
                            Label("지도에서 보기", systemImage: "map")
                                .frame(maxWidth: .infinity)
                        }
                        .buttonStyle(.borderedProminent)

                        Button {
                            sharePlace(place)
                        } label: {
                            Label("공유", systemImage: "square.and.arrow.up")
                                .frame(maxWidth: .infinity)
                        }
                        .buttonStyle(.bordered)
                    }
                }
                .padding(.horizontal)
            }

            Spacer()
        }
        .padding(.vertical)
    }

    func openInMaps(_ place: RecognizedPlace) {
        let mapItem = MKMapItem(
            placemark: MKPlacemark(coordinate: place.coordinate)
        )
        mapItem.name = place.name
        mapItem.openInMaps()
    }

    func sharePlace(_ place: RecognizedPlace) {
        print("📤 \(place.name) 공유")
    }
}

🎯 6. Camera Control 통합

Integrates with iPhone 16's Camera Control button for quick access.

CameraControlIntegration.swift — Camera Control
import SwiftUI

// Camera Control 제스처 핸들링
@Observable
class CameraControlManager {
    var isPressed = false
    var pressStartTime: Date?

    // 길게 누르기 감지
    func handleLongPress() {
        isPressed = true
        pressStartTime = Date()

        // Visual Intelligence 활성화
        activateVisualIntelligence()
    }

    // 버튼 놓기
    func handleRelease() {
        guard let startTime = pressStartTime else { return }

        let duration = Date().timeIntervalSince(startTime)

        if duration > 0.5 {
            // 긴 누르기: 화면 캡처 및 분석
            captureAndAnalyze()
        }

        isPressed = false
        pressStartTime = nil
    }

    func activateVisualIntelligence() {
        print("📸 Visual Intelligence 활성화")
    }

    func captureAndAnalyze() {
        print("🔍 화면 캡처 및 분석 시작")
    }
}

struct CameraControlDemoView: View {
    @State private var manager = CameraControlManager()

    var body: some View {
        VStack {
            Text("Camera Control 버튼을 길게 누르세요")
                .font(.headline)
                .padding()

            // 시뮬레이션용 버튼
            Button {
                manager.handleLongPress()

                Task {
                    try? await Task.sleep(for: .seconds(1))
                    manager.handleRelease()
                }
            } label: {
                Circle()
                    .fill(manager.isPressed ? Color.blue : Color.gray)
                    .frame(width: 80, height: 80)
                    .overlay {
                        Image(systemName: "camera.fill")
                            .font(.title)
                            .foregroundStyle(.white)
                    }
            }
        }
    }
}

💡 HIG Guidelines

  • 프라이버시 우선: All processing is performed on-device; user data is never sent to a server.
  • 즉각적인 피드백: 분석 시작 시 시각적 피드백 제공
  • 맥락 인식: 현재 상황에 맞는 정보만 표시
  • 명확한 액션: 인식된 정보로 무엇을 할 수 있는지 명확히 제시
  • Camera Control 통합: 길게 누르기로 빠르게 접근 가능

🎯 Practical Usage

  • 쇼핑 앱: 제품 스캔 후 가격 비교 및 구매
  • 번역 앱: 표지판, 메뉴판 실시간 번역
  • Education Apps: 숙제 문제 스캔 및 해설 제공
  • Travel Apps: 랜드마크 인식 후 관광 정보 표시
  • 식물 인식: 식물 종류 및 관리 방법 안내
