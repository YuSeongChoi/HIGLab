✨ User Notifications?

The User Notifications framework manages both local and remote push notifications. It provides powerful features including time/location-based triggers, media attachments, custom actions, and Notification Service Extensions.

💡 Key Features: Local/Remote Notifications · Time/Location Triggers · Media Attachments · Custom Actions · Notification Management · Notification Extension · Settings · Foreground Alerts

🎯 1. 권한 요청

알림 사용 전 먼저 사용자 권한을 요청해야 .

NotificationManager.swift — Permission Request
import UserNotifications
import SwiftUI

@Observable
class NotificationManager: NSObject {
    var authorizationStatus: UNAuthorizationStatus = .notDetermined
    var isAuthorized = false

    override init() {
        super.init()
        UNUserNotificationCenter.current().delegate = self
        checkAuthorizationStatus()
    }

    // 권한 상태 확인
    func checkAuthorizationStatus() {
        UNUserNotificationCenter.current().getNotificationSettings { [weak self] settings in
            Task { @MainActor in
                self?.authorizationStatus = settings.authorizationStatus
                self?.isAuthorized = settings.authorizationStatus == .authorized
            }
        }
    }

    // 권한 요청
    func requestAuthorization() async throws -> Bool {
        let center = UNUserNotificationCenter.current()

        // 알림, 사운드, 배지 권한 요청
        let granted = try await center.requestAuthorization(
            options: [.alert, .sound, .badge, .criticalAlert]
        )

        await MainActor.run {
            isAuthorized = granted
        }

        // 원격 푸시 알림 등록
        if granted {
            await UIApplication.shared.registerForRemoteNotifications()
        }

        return granted
    }
}

⏰ 2. Local Notifications

앱에서 직접 스케줄링하는 알림.

LocalNotifications.swift — Time-Based Notifications
import UserNotifications

extension NotificationManager {
    // 시간 기반 알림
    func scheduleTimeNotification(
        title: String,
        body: String,
        timeInterval: TimeInterval,
        repeats: Bool = false
    ) async throws {
        let content = UNMutableNotificationContent()
        content.title = title
        content.body = body
        content.sound = .default
        content.badge = 1

        // 트리거 설정 (5초 후)
        let trigger = UNTimeIntervalNotificationTrigger(
            timeInterval: timeInterval,
            repeats: repeats
        )

        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: trigger
        )

        try await UNUserNotificationCenter.current().add(request)
    }

    // 캘린더 기반 알림 (매일 오전 9시)
    func scheduleDailyNotification(
        title: String,
        body: String,
        hour: Int,
        minute: Int
    ) async throws {
        let content = UNMutableNotificationContent()
        content.title = title
        content.body = body
        content.sound = .default

        var dateComponents = DateComponents()
        dateComponents.hour = hour
        dateComponents.minute = minute

        let trigger = UNCalendarNotificationTrigger(
            dateMatching: dateComponents,
            repeats: true
        )

        let request = UNNotificationRequest(
            identifier: "daily-\(hour)-\(minute)",
            content: content,
            trigger: trigger
        )

        try await UNUserNotificationCenter.current().add(request)
    }

    // 위치 기반 알림
    func scheduleLocationNotification(
        title: String,
        body: String,
        latitude: Double,
        longitude: Double,
        radius: Double
    ) async throws {
        let content = UNMutableNotificationContent()
        content.title = title
        content.body = body
        content.sound = .default

        let center = CLLocationCoordinate2D(latitude: latitude, longitude: longitude)
        let region = CLCircularRegion(
            center: center,
            radius: radius,
            identifier: UUID().uuidString
        )
        region.notifyOnEntry = true
        region.notifyOnExit = false

        let trigger = UNLocationNotificationTrigger(
            region: region,
            repeats: false
        )

        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: trigger
        )

        try await UNUserNotificationCenter.current().add(request)
    }
}

📷 3. Media Attachments

알림에 이미지, 비디오, 오디오를 첨부.

MediaNotifications.swift — Media Attachments
import UserNotifications

extension NotificationManager {
    // 이미지 첨부
    func scheduleNotificationWithImage(
        title: String,
        body: String,
        imageURL: URL
    ) async throws {
        let content = UNMutableNotificationContent()
        content.title = title
        content.body = body
        content.sound = .default

        // 이미지 다운로드 및 첨부
        let (data, _) = try await URLSession.shared.data(from: imageURL)

        let tempDirectory = FileManager.default.temporaryDirectory
        let fileURL = tempDirectory.appendingPathComponent("notification-image.jpg")

        try data.write(to: fileURL)

        let attachment = try UNNotificationAttachment(
            identifier: "image",
            url: fileURL,
            options: nil
        )

        content.attachments = [attachment]

        let trigger = UNTimeIntervalNotificationTrigger(timeInterval: 5, repeats: false)
        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: trigger
        )

        try await UNUserNotificationCenter.current().add(request)
    }

    // 비디오 첨부
    func scheduleNotificationWithVideo(videoURL: URL) async throws {
        let content = UNMutableNotificationContent()
        content.title = "새 비디오"
        content.body = "비디오를 확인하세요"

        let attachment = try UNNotificationAttachment(
            identifier: "video",
            url: videoURL,
            options: [UNNotificationAttachmentOptionsTypeHintKey: "public.mpeg-4"]
        )

        content.attachments = [attachment]

        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: nil
        )

        try await UNUserNotificationCenter.current().add(request)
    }
}

🎬 4. 커스텀 액션 (Actions)

Add buttons to perform actions directly from notifications.

NotificationActions.swift — Custom Actions
import UserNotifications

extension NotificationManager {
    // 액션 카테고리 등록
    func registerNotificationCategories() {
        // 메시지 알림 액션
        let replyAction = UNTextInputNotificationAction(
            identifier: "REPLY_ACTION",
            title: "답장",
            options: [],
            textInputButtonTitle: "보내기",
            textInputPlaceholder: "메시지를 입력하세요"
        )

        let likeAction = UNNotificationAction(
            identifier: "LIKE_ACTION",
            title: "👍",
            options: [.foreground]
        )

        let deleteAction = UNNotificationAction(
            identifier: "DELETE_ACTION",
            title: "삭제",
            options: [.destructive]
        )

        let messageCategory = UNNotificationCategory(
            identifier: "MESSAGE_CATEGORY",
            actions: [replyAction, likeAction, deleteAction],
            intentIdentifiers: [],
            options: []
        )

        // 리마인더 알림 액션
        let completeAction = UNNotificationAction(
            identifier: "COMPLETE_ACTION",
            title: "완료",
            options: []
        )

        let snoozeAction = UNNotificationAction(
            identifier: "SNOOZE_ACTION",
            title: "10분 뒤 알림",
            options: []
        )

        let reminderCategory = UNNotificationCategory(
            identifier: "REMINDER_CATEGORY",
            actions: [completeAction, snoozeAction],
            intentIdentifiers: [],
            options: []
        )

        UNUserNotificationCenter.current().setNotificationCategories([
            messageCategory,
            reminderCategory
        ])
    }

    // 액션이 있는 알림 전송
    func scheduleActionableNotification() async throws {
        let content = UNMutableNotificationContent()
        content.title = "새 메시지"
        content.body = "John이 메시지를 보냈습니다"
        content.categoryIdentifier = "MESSAGE_CATEGORY"
        content.sound = .default

        let trigger = UNTimeIntervalNotificationTrigger(timeInterval: 5, repeats: false)
        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: trigger
        )

        try await UNUserNotificationCenter.current().add(request)
    }
}

🎯 5. Delegate 처리

알림 응답과 포그라운드 알림 handling.

NotificationDelegate.swift — Delegate Implementation
import UserNotifications

extension NotificationManager: UNUserNotificationCenterDelegate {
    // 포그라운드에서 알림 표시
    func userNotificationCenter(
        _ center: UNUserNotificationCenter,
        willPresent notification: UNNotification
    ) async -> UNNotificationPresentationOptions {
        // 앱이 활성화 상태일 때도 알림 표시
        return [.banner, .sound, .badge]
    }

    // 알림 응답 처리 (탭하거나 액션 선택)
    func userNotificationCenter(
        _ center: UNUserNotificationCenter,
        didReceive response: UNNotificationResponse
    ) async {
        let userInfo = response.notification.request.content.userInfo
        let actionIdentifier = response.actionIdentifier

        switch actionIdentifier {
        case "REPLY_ACTION":
            if let textResponse = response as? UNTextInputNotificationResponse {
                let replyText = textResponse.userText
                // 답장 처리
                print("답장: \(replyText)")
            }

        case "LIKE_ACTION":
            // 좋아요 처리
            print("좋아요 클릭")

        case "DELETE_ACTION":
            // 삭제 처리
            print("삭제 클릭")

        case "COMPLETE_ACTION":
            // 완료 처리
            print("완료 클릭")

        case "SNOOZE_ACTION":
            // 10분 뒤 다시 알림
            try? await scheduleTimeNotification(
                title: "리마인더",
                body: "다시 알려드립니다",
                timeInterval: 600
            )

        case UNNotificationDefaultActionIdentifier:
            // 알림 탭 (기본 액션)
            print("알림 탭됨")

        case UNNotificationDismissActionIdentifier:
            // 알림 무시
            print("알림 무시됨")

        default:
            break
        }
    }
}

📱 6. 알림 관리

예약된 알림을 조회, 수정, 삭제.

NotificationManagement.swift — Notification Management
import UserNotifications

extension NotificationManager {
    // 대기 중인 알림 목록
    func getPendingNotifications() async -> [UNNotificationRequest] {
        await UNUserNotificationCenter.current().pendingNotificationRequests()
    }

    // 전달된 알림 목록
    func getDeliveredNotifications() async -> [UNNotification] {
        await UNUserNotificationCenter.current().deliveredNotifications()
    }

    // 특정 알림 취소
    func cancelNotification(withIdentifier identifier: String) {
        UNUserNotificationCenter.current().removePendingNotificationRequests(
            withIdentifiers: [identifier]
        )
    }

    // 모든 알림 취소
    func cancelAllNotifications() {
        UNUserNotificationCenter.current().removeAllPendingNotificationRequests()
    }

    // 전달된 알림 제거
    func removeDeliveredNotification(withIdentifier identifier: String) {
        UNUserNotificationCenter.current().removeDeliveredNotifications(
            withIdentifiers: [identifier]
        )
    }

    // 배지 숫자 초기화
    func clearBadge() {
        Task { @MainActor in
            UIApplication.shared.applicationIconBadgeNumber = 0
        }
    }
}

📱 SwiftUI Integration

NotificationDemoView.swift — Complete Demo
import SwiftUI

struct NotificationDemoView: View {
    @State private var manager = NotificationManager()
    @State private var pendingNotifications: [UNNotificationRequest] = []

    var body: some View {
        NavigationStack {
            List {
                Section("권한") {
                    if manager.isAuthorized {
                        Label("알림 허용됨", systemImage: "checkmark.circle.fill")
                            .foregroundStyle(.green)
                    } else {
                        Button("알림 권한 요청") {
                            Task {
                                try? await manager.requestAuthorization()
                            }
                        }
                    }
                }

                Section("로컬 알림") {
                    Button("5초 후 알림") {
                        Task {
                            try? await manager.scheduleTimeNotification(
                                title: "테스트 알림",
                                body: "5초가 지났습니다",
                                timeInterval: 5
                            )
                        }
                    }

                    Button("매일 오전 9시 알림") {
                        Task {
                            try? await manager.scheduleDailyNotification(
                                title: "좋은 아침!",
                                body: "오늘도 좋은 하루 되세요",
                                hour: 9,
                                minute: 0
                            )
                        }
                    }

                    Button("액션이 있는 알림") {
                        Task {
                            manager.registerNotificationCategories()
                            try? await manager.scheduleActionableNotification()
                        }
                    }
                }

                Section("예약된 알림 (\(pendingNotifications.count)개)") {
                    ForEach(pendingNotifications, id: \.identifier) { notification in
                        VStack(alignment: .leading) {
                            Text(notification.content.title)
                                .font(.headline)
                            Text(notification.content.body)
                                .font(.caption)
                                .foregroundStyle(.secondary)
                        }
                        .swipeActions {
                            Button("삭제", role: .destructive) {
                                manager.cancelNotification(withIdentifier: notification.identifier)
                                loadPendingNotifications()
                            }
                        }
                    }
                }

                Section {
                    Button("모든 알림 취소", role: .destructive) {
                        manager.cancelAllNotifications()
                        loadPendingNotifications()
                    }

                    Button("배지 초기화") {
                        manager.clearBadge()
                    }
                }
            }
            .navigationTitle("User Notifications")
            .task {
                loadPendingNotifications()
            }
            .refreshable {
                loadPendingNotifications()
            }
        }
    }

    func loadPendingNotifications() {
        Task {
            pendingNotifications = await manager.getPendingNotifications()
        }
    }
}

💡 HIG Guidelines

  • 시의적절성: Only send notifications for important, time-sensitive information
  • 명확성: 알림 내용을 간결하고 명확하게 작성
  • 액션: 알림에서 바로 수행 가능한 의미 있는 액션 제공
  • 타이밍: 사용자가 방해받지 않는 적절한 시간에 알림 전송
  • 개인화: 사용자 행동과 선호도를 반영한 알림
  • 제어: Provide settings for users to control notification frequency and type

🎯 Practical Usage

  • 리마인더 앱: 시간/위치 기반 할 일 알림
  • Messaging Apps: 새 메시지 알림 + 빠른 답장
  • Fitness Apps: 운동 시간 알림 + 미디어 첨부
  • 배달 앱: 주문 상태 업데이트 알림
  • News Apps: 속보 알림 with 이미지
