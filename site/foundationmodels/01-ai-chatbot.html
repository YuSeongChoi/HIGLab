Foundation Models를 활용해 프라이버시를 보호하는 AI 앱을 만듭니다. 모든 처리가 기기 내에서 이루어집니다.

✨ Foundation Models 특징

• 프라이버시 보호 (데이터가 기기를 떠나지 않음)
• 오프라인 지원 (네트워크 불필요)
• Swift 네이티브 async/await API

💬 기본 사용법 (LanguageModelSession)

import FoundationModels

// 세션 생성 (시스템 프롬프트 설정)
let session = LanguageModelSession(
    instructions: "친절한 AI 어시스턴트입니다"
)

// 응답 요청
let response = try await session.respond(to: "안녕하세요")
print(response.content)

🔄 스트리밍 응답

💬 SwiftUI 챗봇 구현

import SwiftUI
import FoundationModels

struct ChatView: View {
    @State private var messages: [Message] = []
    @State private var inputText = ""
    @State private var isGenerating = false
    @State private var streamingText = ""

    // LanguageModelSession 사용
    @State private var session = LanguageModelSession(
        instructions: "친절한 AI 어시스턴트입니다. 한국어로 답변하세요."
    )

    var body: some View {
        VStack {
            ScrollView {
                ForEach(messages) { message in
                    MessageBubble(message: message)
                }
                // 스트리밍 중인 응답 표시
                if !streamingText.isEmpty {
                    MessageBubble(message: Message(role: .assistant, content: streamingText))
                }
            }

            HStack {
                TextField("메시지 입력", text: $inputText)
                    .textFieldStyle(.roundedBorder)

                Button("전송") {
                    sendMessage()
                }
                .disabled(isGenerating || inputText.isEmpty)
            }
            .padding()
        }
    }

    func sendMessage() {
        let userMessage = Message(role: .user, content: inputText)
        messages.append(userMessage)
        let prompt = inputText
        inputText = ""
        isGenerating = true
        streamingText = ""

        Task {
            do {
                // 스트리밍 응답
                let stream = session.streamResponse(to: prompt)
                for try await partial in stream {
                    streamingText = partial.outputSoFar
                }
                // 완료 후 메시지 배열에 추가
                messages.append(Message(role: .assistant, content: streamingText))
                streamingText = ""
            } catch {
                print("오류: \(error)")
            }
            isGenerating = false
        }
    }
}

struct Message: Identifiable {
    let id = UUID()
    let role: MessageRole
    var content: String
}

enum MessageRole {
    case user, assistant
}

⚙️ 모델 설정 & 에러 처리

do {
    // Temperature 조정 (0.0 = 일관성, 1.0 = 창의성)
    let result = try await model.generate(
        prompt: prompt,
        temperature: 0.7,
        maxTokens: 500
    )
} catch let error as FoundationModelsError {
    switch error {
    case .modelNotAvailable:
        print("모델이 다운로드되지 않았습니다")
    case .insufficientMemory:
        print("메모리가 부족합니다")
    default:
        print("오류 발생: \(error)")
    }
}
